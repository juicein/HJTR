// ========================== // ⚙️ 配置 // ========================== const TILE_SIZE = 256; const MODE = "unmined"; // "dynmap" const MIN_ZOOM = -6; const MAX_ZOOM = 0; const container = document.querySelector(".tile-container"); // ========================== // 🌍 状态（兼容你原变量） // ========================== let state = { centerX: 0, centerZ: 0, zoom: zoomLevel || -1 }; // ========================== // 🧠 获取 Tile URL（双格式支持） // ========================== function getTileURL(z, x, y) { if (MODE === "unmined") { return `/map/tiles/zoom.${z}/${x}/${y}/tile.${x}.${y}.webp`; } if (MODE === "dynmap") { const chunkX = Math.floor(x / 32); const chunkY = Math.floor(y / 32); return `/map/tiles/world/flat/${chunkX}_${chunkY}/${x}_${y}.png`; } } // ========================== // 🧠 渲染 Tile（核心） // ========================== function renderTiles() { container.innerHTML = ""; const width = container.clientWidth; const height = container.clientHeight; const tilesX = Math.ceil(width / TILE_SIZE) + 2; const tilesZ = Math.ceil(height / TILE_SIZE) + 2; const startTileX = Math.floor(state.centerX / TILE_SIZE); const startTileZ = Math.floor(state.centerZ / TILE_SIZE); for (let dx = -tilesX; dx < tilesX; dx++) { for (let dz = -tilesZ; dz < tilesZ; dz++) { const tileX = startTileX + dx; const tileZ = startTileZ + dz; const img = document.createElement("img"); img.className = "tile"; const screenX = (tileX * TILE_SIZE - state.centerX + width / 2); const screenZ = (tileZ * TILE_SIZE - state.centerZ + height / 2); img.style.left = screenX + "px"; img.style.top = screenZ + "px"; img.src = getTileURL(state.zoom, tileX, tileZ); img.onerror = () => img.remove(); container.appendChild(img); } } } // ========================== // 🖱️ 拖动（兼容你变量） // ========================== container.addEventListener("mousedown", (e) => { isDragging = true; dragStartClientX = e.clientX; dragStartClientY = e.clientY; dragStartX = state.centerX; dragStartZ = state.centerZ; }); window.addEventListener("mouseup", () => { isDragging = false; }); window.addEventListener("mousemove", (e) => { if (!isDragging) return; const dx = e.clientX - dragStartClientX; const dz = e.clientY - dragStartClientY; state.centerX = dragStartX - dx; state.centerZ = dragStartZ - dz; renderTiles(); }); // ========================== // 🔍 滚轮缩放（核心） // ========================== container.addEventListener("wheel", (e) => { let oldZoom = state.zoom; if (e.deltaY < 0) { state.zoom++; } else { state.zoom--; } state.zoom = Math.max(MIN_ZOOM, Math.min(MAX_ZOOM, state.zoom)); let scale = Math.pow(2, state.zoom - oldZoom); state.centerX *= scale; state.centerZ *= scale; zoomLevel = state.zoom; // 同步你的变量 renderTiles(); }); // ========================== // 📱 双指缩放（你已有变量） // ========================== container.addEventListener("touchstart", (e) => { if (e.touches.length === 2) { isZooming = true; const dx = e.touches[0].clientX - e.touches[1].clientX; const dy = e.touches[0].clientY - e.touches[1].clientY; initialPinchDistance = Math.sqrt(dx*dx + dy*dy); initialZoomLevel = state.zoom; } }); container.addEventListener("touchmove", (e) => { if (!isZooming || e.touches.length !== 2) return; const dx = e.touches[0].clientX - e.touches[1].clientX; const dy = e.touches[0].clientY - e.touches[1].clientY; currentPinchDistance = Math.sqrt(dx*dx + dy*dy); let delta = (currentPinchDistance - initialPinchDistance) / 200; let newZoom = Math.round(initialZoomLevel + delta); newZoom = Math.max(MIN_ZOOM, Math.min(MAX_ZOOM, newZoom)); if (newZoom !== state.zoom) { state.zoom = newZoom; renderTiles(); } }); container.addEventListener("touchend", () => { isZooming = false; }); // ========================== // 🚀 初始化 // ========================== renderTiles(); window.addEventListener("resize", renderTiles);
